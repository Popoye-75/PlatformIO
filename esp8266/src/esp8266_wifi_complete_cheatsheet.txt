#include <ESP8266WiFi.h> // ESP8266 WiFi library include karta hai

const char* ssid = “YOUR_WIFI_NAME”; // WiFi router ka naam const char*
password = “YOUR_WIFI_PASS”; // WiFi password

Serial.begin(9600); // Serial monitor start karta hai debugging ke liye

WiFi.begin(ssid, password); // Router se WiFi connection start karta hai

  SECTION 9: SIGNAL STRENGTH GUIDE
  ----------------------------------

-30 dBm → Excellent signal -50 dBm → Good signal -70 dBm → Weak signal
-80 dBm → Very weak / unstable

  --------------------------------
  SECTION 10: COMMON WIFI ERRORS
  --------------------------------

1)  ESP8266 5GHz WiFi support nahi karta Sirf 2.4GHz WiFi chalega

2)  Power supply unstable ho to WiFi disconnect hota hai

3)  Wrong password se WL_CONNECT_FAILED milta hai

4)  Weak signal se connection unstable ho jata hai
